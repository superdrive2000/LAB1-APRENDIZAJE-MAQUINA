Visualización en Altair

Visaulizacion 1: Pie Graph Distribución por rango de edad

Code 
edades = personas["rango_edad"].value_counts().reset_index()
edades.columns = ["rango_edad", "cantidad"]

alt.Chart(edades).mark_arc().encode(
    theta=alt.Theta(field="cantidad", type="quantitative"),
    color=alt.Color(field="rango_edad", type="nominal"),
    tooltip=["rango_edad", "cantidad"]
).properties(
    title="Distribución de desaparecidos por rango de edad",
    width=400, height=400
)

Visaulizacion 2:

Code 
data_motivos = personas[["motivo_desaparicion"]].copy()
motivos_top = data_motivos["motivo_desaparicion"].value_counts().nlargest(10).index.tolist()
data_motivos = data_motivos[data_motivos["motivo_desaparicion"].isin(motivos_top)]

alt.Chart(data_motivos).mark_bar().encode(
    y=alt.Y("motivo_desaparicion:N", sort='-x', title="Motivo de desaparición"),
    x=alt.X("count()", title="Cantidad"),
    color="motivo_desaparicion:N",
    tooltip=["motivo_desaparicion", "count()"]
).properties(
    title="Motivos más frecuentes de desaparición (Top 10)",
    width=500, height=300
)
